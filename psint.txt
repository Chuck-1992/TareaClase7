
// Validar si una dato es de tipo entero
Funcion resultado <- validar_tipo_entero(texto)
    Definir i Como Entero
    Definir caracter Como Cadena
	
    resultado <- Verdadero
	
    Si Longitud(texto) = 0 Entonces
        resultado <- Falso
    SiNo
        Para i <- 1 Hasta Longitud(texto) Con Paso 1 Hacer
            caracter <- Subcadena(texto, i, i)
			
            Si caracter < "0" O caracter > "9" Entonces
                resultado <- Falso
            FinSi
        FinPara
    FinSi
FinFuncion

// Validar si una dato es de tipo decimal
Funcion resultado <- validar_tipo_real(texto)
    Definir i, puntos Como Entero
    Definir caracter Como Cadena
	
    resultado <- Verdadero
    puntos <- 0
	
    Si Longitud(texto) = 0 Entonces
        resultado <- Falso
    SiNo
        Para i <- 1 Hasta Longitud(texto) Con Paso 1 Hacer
            caracter <- Subcadena(texto, i, i)
			
            Si caracter = "." Entonces
                puntos <- puntos + 1
                Si puntos > 1 Entonces
                    resultado <- Falso
                FinSi
            SiNo
                Si caracter < "0" O caracter > "9" Entonces
                    resultado <- Falso
                FinSi
            FinSi
        FinPara
    FinSi
	
    Si texto = "." Entonces
        resultado <- Falso
    FinSi
FinFuncion


// Solicitar al usuario un numero entero
Funcion valor <- pedir_entero_rango(mensaje, minimo, maximo)
	
	Definir texto Como Cadena
	Definir numero Como Entero
	Definir correcto Como Logico
	
	correcto <- Falso
	
	Repetir
		
		Escribir mensaje
		Leer texto
		
		Si validar_tipo_entero(texto) Entonces
			
			numero <- ConvertirANumero(texto)
			
			Si numero >= minimo Y numero <= maximo Entonces
				valor <- numero
				correcto <- Verdadero
			SiNo
				Escribir "Error: el número debe estar entre ", minimo, " y ", maximo, "."
			FinSi
			
		SiNo
			Escribir "Error: debe ingresar un número entero válido."
		FinSi
		
	Hasta Que correcto = Verdadero
	
FinFuncion

Algoritmo Menu_Condicionales_Operadores_Relacionales
	
	Definir opMenu Como Entero
	Definir edad Como Entero
	Definir nota Como Real
	Definir cadena1, cadena2 Como Cadena
	Definir rango, clasificacion, resultado Como Cadena
	Definir salir_programa Como Logico
	
	salir_programa <- Falso
	
	Escribir "Escriba la opción 5 para cerrar el programa."
	
	Mientras salir_programa = Falso Hacer
		
		Escribir ""
		Escribir "************ MENÚ DE OPCIONES ***************"
		Escribir " 1. Calcular rango de edad"
		Escribir " 2. Clasificar nota"
		Escribir " 3. Comparar cadenas"
		Escribir " 4. Limpiar consola"
		Escribir " 5. Salir"
		Escribir "*********************************************"
		Escribir ""
		
		opMenu <- pedir_entero_rango("Ingrese una opción del 1 al 5:", 1, 5)
		
		Si opMenu = 5 Entonces
			
			Escribir "Programa finalizado. Gracias por usar el menú."
			salir_programa <- Verdadero
			
		SiNo
			
			Si opMenu = 1 Entonces
				
				edad <- pedir_entero_rango("Ingrese su edad:", 0, 150)
				
				Si edad < 12 Entonces
					rango <- "Niño/a"
				SiNo
					Si edad < 18 Entonces
						rango <- "Adolescente"
					SiNo
						Si edad < 60 Entonces
							rango <- "Adulto"
						SiNo
							rango <- "Adulto mayor"
						FinSi
					FinSi
				FinSi
				
				Escribir ""
				Escribir "Calcular rango de edad:"
				Escribir "Edad ingresada: ", edad
				Escribir "Rango: ", rango
				
			SiNo
				
				Si opMenu = 2 Entonces
					
					nota <- pedir_real_rango("Ingrese una nota entre 0 y 10:", 0, 10)
					
					Si nota >= 9 Entonces
						clasificacion <- "Excelente"
					SiNo
						Si nota >= 7 Entonces
							clasificacion <- "Aprobado"
						SiNo
							Si nota >= 5 Entonces
								clasificacion <- "Recuperación"
							SiNo
								clasificacion <- "Reprobado"
							FinSi
						FinSi
					FinSi
					
					Escribir ""
					Escribir "Clasificar nota:"
					Escribir "Nota: ", nota
					Escribir "Clasificación: ", clasificacion
					
				SiNo
					
					Si opMenu = 3 Entonces
						
						Repetir
							Escribir "Ingrese la primera cadena:"
							Leer cadena1
							
							Si cadena1 = "" Entonces
								Escribir "Error: la primera cadena no puede estar vacía."
							FinSi
							
						Hasta Que cadena1 <> ""
						
						Repetir
							Escribir "Ingrese la segunda cadena:"
							Leer cadena2
							
							Si cadena2 = "" Entonces
								Escribir "Error: la segunda cadena no puede estar vacía."
							FinSi
							
						Hasta Que cadena2 <> ""
						
						Si cadena1 = cadena2 Entonces
							resultado <- "Son exactamente iguales"
						SiNo
							Si Longitud(cadena1) > Longitud(cadena2) Entonces
								resultado <- "La primera es más larga"
							SiNo
								Si Longitud(cadena1) < Longitud(cadena2) Entonces
									resultado <- "La segunda es más larga"
								SiNo
									resultado <- "Misma longitud, pero distintas"
								FinSi
							FinSi
						FinSi
						
						Escribir ""
						Escribir "Comparar cadenas:"
						Escribir "Cadena 1: ", cadena1
						Escribir "Longitud 1: ", Longitud(cadena1)
						Escribir "Cadena 2: ", cadena2
						Escribir "Longitud 2: ", Longitud(cadena2)
						Escribir "Resultado: ", resultado
						
					SiNo
						
						Si opMenu = 4 Entonces
							Limpiar Pantalla
							Escribir "Consola limpiada correctamente."
						FinSi
						
					FinSi
					
				FinSi
				
			FinSi
			
		FinSi
		
	FinMientras
	
FinAlgoritmo
